265

268  uint32_t total = __popc(mask); // Number of active threads
269  unsigned int rank = __popc(mask & lanemask_lt()); // Rank of current thread
270  int leader =
271  __ffs(mask) -
272  1; // Position of least sig. bit -> whether thread is min active
273

275  uint64_t c_offset = q_id * PADDING_SIZE;

277  CounterT alloc;
278  if (rank == 0) {

280  total); // Get + increment index to write to
281  }
282 
283  alloc =
284  __shfl_sync(mask, alloc, leader); // copy alloc to all active threads
285  assert(alloc + total <= capacity);

292  CounterT end_count_old =
293  atomicAdd((CounterT*)(end_count + c_offset),
294  total); // Update number of items in queue
295  CounterT end_max_old =
296  atomicMax((CounterT*)(end_max + c_offset),
297  alloc + total); // Update maximum position of item in queue

301  CounterT end_count_new = end_count_old + total;

303  if (end_count_new == end_max_new) { // If queue is full up to last
