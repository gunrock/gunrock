110  // Main loop that goes over all the neighbors and finds the maximum or

112  for (edge_t e = start_edge; e < start_edge + num_neighbors; ++e) {
113  vertex_t u = G.get_destination_vertex(e);
114

121  if (randoms[vertex] >= randoms[u])

123  }
